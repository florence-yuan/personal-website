Animations



    const figure = document.querySelector('.moving-figure');

    let prevProgress = 0, prevFrameCount = 0;
    const bgFrameCnt = 2;
    const figureFrames = {
        frameCount: 0
    };

    gsap.to(figureFrames, {
        scrollTrigger: {
            scrub: 0.5,
            start: '30% center',
            // frameCount: figureFrames.frameCount + 1,
            onUpdate: self => {
                if (Math.abs(self.progress - prevProgress) < 0.1 && self.progress !== 0)
                    return;

                figureFrames.frameCount = Math.min(bgFrameCnt - 1, Math.ceil(self.progress * bgFrameCnt));
                if (figureFrames.frameCount === prevFrameCount)
                    return;

                gsap.set(figure, { backgroundPositionY: `${100 * figureFrames.frameCount}%`});
                
                prevProgress = self.progress;
                prevFrameCount = figureFrames.frameCount;
            }
        },
        ease: 'none',
    });

    gsap.fromTo('.hero-section .section__heading',
        {
            opacity: 1,
        },
        {
            scrollTrigger: {
                start: 'center center',
                end: 'bottom',
                trigger: '.hero-section .section__heading',
                scrub: 0.5,
            },
            opacity: 0
        }
    );

    const introHeadings = document.querySelectorAll('.intro__heading');
    introHeadings.forEach(heading => {
        gsap.fromTo(heading,
            { opacity: 0, scale: 0.95, y: 20 },
            {
                scrollTrigger: {
                    trigger: heading,
                    start: "center center",
                },
                opacity: 1,
                scale: 1,
                y: 0
            }
        )
    });
